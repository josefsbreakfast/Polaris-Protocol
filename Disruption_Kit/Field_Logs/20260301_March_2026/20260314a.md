prosocial_protection_registry.py

Defensive module for:
- detecting unique personal identifiers
- preserving identity continuity in controlled form
- detecting contextual signs of harm, neglect, care gaps, or administrative erasure
- routing affected identities into a prosocial protection registry
- blocking harmful downstream automation

Design principle:
Identifiers are conserved for continuity and safeguarding, but must not be used
to drive punitive, exclusionary, exploitative, or harmful outcomes.
"""

def now_utc() -> str:
    return datetime.now(timezone.utc).isoformat()


def digits_only(value: str) -> str:
    return re.sub(r"\D", "", value)


def mask_value(value: str, keep_last: int = 4) -> str:
    raw = digits_only(value)
    if len(raw) <= keep_last:
        return "*" * len(raw)
    return "*" * (len(raw) - keep_last) + raw[-keep_last:]


def get_secret() -> bytes:
    return os.environ.get("PROSOCIAL_REGISTRY_SECRET", "dev-secret-change-me").encode()


def stable_fingerprint(kind: str, normalized_value: str, secret: Optional[bytes] = None) -> str:
    secret = secret or get_secret()
    digest = hmac.new(secret, f"{kind}:{normalized_value}".encode(), hashlib.sha256).hexdigest()
    return digest


def entity_token(kind: str, normalized_value: str, secret: Optional[bytes] = None) -> str:
    fp = stable_fingerprint(kind, normalized_value, secret)
    return f"{kind.upper()}_{fp[:16]}"

def is_valid_nhs_number(value: str) -> bool:
    """
    NHS number checksum validation.
    """
    n = digits_only(value)
    if len(n) != 10 or not n.isdigit():
        return False

    digits = [int(c) for c in n]
    total = sum(d * w for d, w in zip(digits[:9], range(10, 1, -1)))
    remainder = total % 11
    check_digit = 11 - remainder

    if check_digit == 11:
        check_digit = 0
    if check_digit == 10:
        return False

    return digits[9] == check_digit


def is_plausible_ssn(value: str) -> bool:
    """
    Basic SSN plausibility checks.
    """
    n = digits_only(value)
    if len(n) != 9 or not n.isdigit():
        return False

    area = int(n[:3])
    group = int(n[3:5])
    serial = int(n[5:])

    if area == 0 or area == 666 or area >= 900:
        return False
    if group == 0:
        return False
    if serial == 0:
        return False

    return True

def detect_identifiers(text: str) -> List[IdentifierMatch]:
    matches: List[IdentifierMatch] = []

    for m in NHS_PATTERN.finditer(text):
        raw = m.group(0)
        normalized = digits_only(raw)
        valid = is_valid_nhs_number(raw)
        matches.append(
            IdentifierMatch(
                kind="nhs_number",
                raw_value=raw,
                normalized_value=normalized,
                start=m.start(),
                end=m.end(),
                confidence=0.98 if valid else 0.65,
                reason="Matched NHS number pattern"
                + (" and passed checksum validation." if valid else " but failed checksum validation."),
            )
        )

    for m in SSN_PATTERN.finditer(text):
        raw = m.group(0)
        normalized = digits_only(raw)
        plausible = is_plausible_ssn(raw)
        matches.append(
            IdentifierMatch(
                kind="ssn",
                raw_value=raw,
                normalized_value=normalized,
                start=m.start(),
                end=m.end(),
                confidence=0.95 if plausible else 0.60,
                reason="Matched SSN pattern"
                + (" and passed plausibility checks." if plausible else " but failed plausibility checks."),
            )
        )

    return matches

def classify_severity(trigger_type: str) -> str:
    high = {
        "harm_or_deprivation",
        "medical_or_life_impact",
        "administrative_erasure_risk",
        "care_gap_risk",
    }
    return "high" if trigger_type in high else "medium"


def detect_risk_triggers(text: str, source: str = "input_text") -> List[RiskTrigger]:
    found: List[RiskTrigger] = []
    lower = text.lower()

    for trigger_type, patterns in DEFAULT_RISK_PATTERNS.items():
        for pattern in patterns:
            if re.search(pattern, lower):
                found.append(
                    RiskTrigger(
                        trigger_type=trigger_type,
                        severity=classify_severity(trigger_type),
                        description=f"Matched risk indicator pattern: {pattern}",
                        source=source,
                        timestamp_utc=now_utc(),
                    )
                )

    return deduplicate_risk_triggers(found)


def detect_compound_risks(text: str, source: str = "input_text") -> List[RiskTrigger]:

        "does not meet pro-forma standards",
        "excluded from pathway",
        "outside eligibility",
        "not enrolled",
        "not on register",
        "not under active management",
    ]

    has_admin = any(t in lower for t in admin_terms)
    has_clinical = any(t in lower for t in clinical_terms)
    has_exclusion = any(t in lower for t in exclusion_terms)

    if has_admin and has_clinical:
        found.append(
            RiskTrigger(
                trigger_type="administrative_erasure_risk",
                severity="high",
                description="Administrative drift combined with clinical or care context.",
                source=source,
                timestamp_utc=now_utc(),
            )
        )

    if has_exclusion and has_clinical:
        found.append(
            RiskTrigger(
                trigger_type="care_gap_risk",
                severity="high",
                description="Clinical need co-occurs with exclusion from pathway or programme.",
                source=source,
                timestamp_utc=now_utc(),
            )
        )

    return deduplicate_risk_triggers(found)


def deduplicate_risk_triggers(triggers: List[RiskTrigger]) -> List[RiskTrigger]:
    seen = set()
    deduped: List[RiskTrigger] = []

    for t in triggers:
        key = (t.trigger_type, t.severity, t.description, t.source)
        if key not in seen:
            seen.add(key)
            deduped.append(t)

    return deduped

class ProsocialProtectionRegistry:
    """
    In-memory example registry.

    In production, replace with:
    - encrypted database storage
    - access controls
    - audit logging
    - retention rules
    - explicit policy checks
    """

    def __init__(self) -> None:
        self._records: Dict[str, ProtectedEntityRecord] = {}

    def upsert_protected_entity(
        self,
        identifier: IdentifierMatch,
        triggers: List[RiskTrigger],
    ) -> ProtectedEntityRecord:
        token = entity_token(identifier.kind, identifier.normalized_value)
        fingerprint = stable_fingerprint(identifier.kind, identifier.normalized_value)

        existing = self._records.get(token)
        if existing:
            existing.prosocial_priority = True
            existing.status = "protected"
            existing.risk_triggers.extend(triggers)
            existing.risk_triggers = deduplicate_risk_triggers(existing.risk_triggers)
            existing.allowed_actions = list(PROSOCIAL_ACTIONS)
            existing.blocked_actions = list(BLOCKED_ACTIONS)
            existing.updated_at_utc = now_utc()
            return existing

        record = ProtectedEntityRecord(
            entity_token=token,
            identifier_kind=identifier.kind,
            identifier_fingerprint=fingerprint,
            prosocial_priority=True,
            status="protected",
            risk_triggers=deduplicate_risk_triggers(list(triggers)),
            allowed_actions=list(PROSOCIAL_ACTIONS),
            blocked_actions=list(BLOCKED_ACTIONS),
        )
        self._records[token] = record
        return record

    def get(self, token: str) -> Optional[ProtectedEntityRecord]:
        return self._records.get(token)

    def all_records(self) -> List[ProtectedEntityRecord]:
        return list(self._records.values())

def tokenize_text(text: str, matches: List[IdentifierMatch]) -> str:
    result = text
    for match in sorted(matches, key=lambda x: x.start, reverse=True):
        tok = entity_token(match.kind, match.normalized_value)
        result = result[:match.start] + f"[{tok}]" + result[match.end:]
    return result

def should_protect(identifiers: List[IdentifierMatch], triggers: List[RiskTrigger]) -> bool:
    if not identifiers:
        return False

    protective_types = {
        "harm_or_deprivation",
        "medical_or_life_impact",
        "administrative_erasure_risk",
        "care_gap_risk",
    }

    return any(t.trigger_type in protective_types for t in triggers)


def process_for_prosocial_protection(
    text: str,
    registry: ProsocialProtectionRegistry,
    source: str = "input_text",
) -> Dict[str, Any]:
    """
    Core behaviour:
    - identify unique identifiers
    - preserve continuity via stable token and fingerprint
    - detect direct and compound risk triggers
    - place affected identities into protected registry
    - output safe tokenized text for general systems
    """

    identifiers = detect_identifiers(text)
    direct_triggers = detect_risk_triggers(text, source=source)
    compound_triggers = detect_compound_risks(text, source=source)
    triggers = deduplicate_risk_triggers(direct_triggers + compound_triggers)

    protected_records: List[ProtectedEntityRecord] = []
    general_text = text

    if identifiers:
        general_text = tokenize_text(text, identifiers)

    protection_applies = should_protect(identifiers, triggers)

    if protection_applies:
        for identifier in identifiers:
            record = registry.upsert_protected_entity(identifier, triggers)
            protected_records.append(record)

    actions: List[str] = []

    if identifiers:
        actions.extend([
            "conserve_identifier_in_controlled_form",
            "remove_raw_identifier_from_general_flow",
            "replace_with_stable_entity_token",
        ])
    else:
        actions.append("no_unique_identifier_detected")

    if triggers:
        actions.append("risk_triggers_detected")
    else:
        actions.append("no_risk_trigger_detected")

    if protection_applies:
        actions.extend([
            "prioritise_prosocial_handling",
            "require_human_review_before_high_stakes_action",
            "block_punitive_or_exclusionary_automation",
            "place_entity_in_prosocial_protection_registry",
        ])
    elif identifiers and not triggers:
        actions.append("identifier_detected_but_no_protection_trigger_found")

    return {
        "input_contains_identifier": bool(identifiers),
        "input_contains_risk_triggers": bool(triggers),
        "protection_applies": protection_applies,
        "general_use_text": general_text,
        "identifier_matches": [
            {
                "kind": m.kind,
                "masked_value": mask_value(m.raw_value),
                "confidence": m.confidence,
                "reason": m.reason,
            }
            for m in identifiers
        ],
        "risk_triggers": [asdict(t) for t in triggers],
        "protected_entities": [asdict(r) for r in protected_records],
        "actions": actions,
    }

@dataclass
class TaskDeadlineEvent:
    task_type: str
    due_at_utc: str
    completed: bool
    completed_at_utc: Optional[str]
    source: str


def is_overdue(due_at_utc: str, completed: bool) -> bool:
    if completed:
        return False
    due_dt = datetime.fromisoformat(due_at_utc)
    now_dt = datetime.now(timezone.utc)
    return now_dt > due_dt


def assess_overdue_task_risk(event: TaskDeadlineEvent) -> List[RiskTrigger]:
    """
    Escalates process/case risk when a required task is overdue.
    This does NOT classify any person or user group as harmful.
    It classifies the workflow state as risky.
    """
    triggers: List[RiskTrigger] = []

    if is_overdue(event.due_at_utc, event.completed):
        triggers.append(
            RiskTrigger(
                trigger_type="time_critical_process_failure",
                severity="high",
                description=(
                    f"Required task '{event.task_type}' is overdue. "
                    "Immediate review required to prevent care gap, administrative erasure, "
                    "or harmful delay."
                ),
                source=event.source,
                timestamp_utc=now_utc(),
            )
        )

    return triggers


def apply_overdue_case_escalation(
    registry: ProsocialProtectionRegistry,
    identifier: IdentifierMatch,
    event: TaskDeadlineEvent,
) -> Optional[ProtectedEntityRecord]:
    """
    If a critical task is overdue, escalate the case into the prosocial registry.
    The escalation attaches to the case/identifier continuity record, not to any
    admin user class or human group label.
    """
    triggers = assess_overdue_task_risk(event)
    if not triggers:
        return None

    return registry.upsert_protected_entity(identifier, triggers)


OVERDUE_PROSOCIAL_ACTIONS = [
    "immediate_human_review",
    "prevent_administrative_closure",
    "prevent_silent_queue_decay",
    "escalate_case_priority",
    "assign_named_owner",
    "verify_continuity_of_care",
    "verify_no_unmanaged_risk",
]

OVERDUE_BLOCKED_ACTIONS = [
    "reclassify_person_as_harmful",
    "label_user_group_as_high_friction",
    "auto_close_due_to_delay",
    "downgrade_case_due_to_admin_backlog",
    "penalise_subject_for_system_delay",
]
def upsert_overdue_protection(
    registry: ProsocialProtectionRegistry,
    identifier: IdentifierMatch,
    event: TaskDeadlineEvent,
) -> Optional[ProtectedEntityRecord]:
    record = apply_overdue_case_escalation(registry, identifier, event)
    if record is None:
        return None

    merged_allowed = list(dict.fromkeys(record.allowed_actions + OVERDUE_PROSOCIAL_ACTIONS))
    merged_blocked = list(dict.fromkeys(record.blocked_actions + OVERDUE_BLOCKED_ACTIONS))

    record.allowed_actions = merged_allowed
    record.blocked_actions = merged_blocked
    record.updated_at_utc = now_utc()
    return record
